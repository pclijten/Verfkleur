Vervang de GOOGLE_SCRIPT_URL in de code met jouw werkelijke Google Apps Script URL om gegevens op te slaan.

Google Apps Script Setup

Stap 1: Maak een Google Spreadsheet

Ga naar Google Sheets en maak een nieuwe spreadsheet met de volgende kolommen:

Timestamp | Locatie | Merk | Kleurcode | Type | Gekocht | Opmerkingen | Muurfoto | Verfblikfoto

Stap 2: Maak Google Apps Script

Ga naar script.google.com en maak een nieuw project.

Stap 3: Voeg de code toe

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.openById('1-aHl3aCOEQVvM7mAe0yzjeQXwMphExr3OKv1azrJ4DA').getActiveSheet();
    
    sheet.appendRow([
      new Date(),
      data.locatie,
      data.merk,
      data.kleurcode,
      data.type,
      data.gekocht,
      data.opmerkingen,
      data.muurfoto,
      data.verfblikfoto
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({status: 'OK'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({status: 'ERROR', message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    const sheet = SpreadsheetApp.openById('1-aHl3aCOEQVvM7mAe0yzjeQXwMphExr3OKv1azrJ4DA').getActiveSheet();
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const rows = data.slice(1);
    
    const result = rows.map(row => {
      const obj = {};
      headers.forEach((header, index) => {
        obj[header.toLowerCase()] = row[index];
      });
      return obj;
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: 'OK', data: result}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({status: 'ERROR', message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
